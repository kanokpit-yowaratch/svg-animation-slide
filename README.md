            item.addEventListener('click', event => {
                // console.log(event.preventDefault);
                const classList = item.className.split(' ');
                if (!classList[2]) {
                    const classSequence = classList[1];
                    focusTheWorld('.' + classSequence);
                }
            });
        });

        function focusTheWorld(world) {
            var notIncludeWorld = document.querySelectorAll('.world:not(' + world + ')');
            // console.log(notIncludeWorld);
            if (notIncludeWorld.length > 0) {
                for (let i = 0; i < notIncludeWorld.length; i++) {
                    notIncludeWorld[i].style.display = "none";
                }
            }
            var notIncludePage = document.querySelectorAll('.page:not(' + world + ')');
            // console.log(notIncludePage);
            if (notIncludePage.length > 0) {
                for (let i = 0; i < notIncludePage.length; i++) {
                    notIncludePage[i].classList.remove("disabled");
                }
            }

            const focusWorld = document.querySelectorAll(world);
            if (focusWorld.length > 0) {
                for (let i = 0; i < focusWorld.length; i++) {
                    // console.log(focusWorld[i].className);
                    if (focusWorld[i].className.includes('world')) {
                        // console.log('word: ', focusWorld[i].className);
                        if (focusWorld[i]) {
                            focusWorld[i].style.display = "block";
                        }
                    }
                    if (focusWorld[i].className.includes('page')) {
                        if (focusWorld[i]) {
                            focusWorld[i].classList.add("disabled");
                        }
                    }
                }
            }
        }
